[tex]\Rightarrow L_{Pt}=(2).(\pi) .(r).(r+t) \\ \Rightarrow L_{Pt}=(2).(\pi).(r).(r+2r) \\ \Rightarrow L_{Pt}=(6).(\pi).(r^{2}) \\ \Rightarrow L_{Pt}=(6) \times (40) \\ \Rightarrow \boxed{L_{Pt}=240\;cm^{2}}[/tex]
